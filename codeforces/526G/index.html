codeforces526G

传送门

题意

给定一棵带有边权的树，多次询问过x号点最多y笔画（可以重叠）能画出的最长连通块，每组询问x y不同，询问独立

思路

题意上看非常一般，感觉没什么特殊的条件
那个x的条件一看就是添乱的。。。

特殊化

先考虑没有x的条件，且
那么显然答案=直径长度
那么如何解决呢
如果考虑把直径拉出来，剩下的就是挂在直径各个节点上的树
首先直径一定仍然在答案中，对于剩下次操作，我们可以每次选取两棵树的各一条链，把链缩成一个点并累加答案
而根据贪心的原理，每次一定取最长的深度
因此我们可以考虑（带权）长链剖分
把每一条剖出来的链塞进一个堆，每次取出最大的即可
不妨来bb证明一波：

  1. 对于一棵树中剖出来的链，在过根的链被选取前一定不会有别的链被选取
  2. 根所在链被选取后剩下的链相当于直接挂在直径上的树，同理
    然而这里其实是有一个疑惑的：
    如果堆顶的两个元素属于同一棵树，显然是不能用一笔画画出来的
    但是实际上画直径的时候扭进来就可以实现了
    （不妨自己画图试试）
要求过x点的做法

大力讨论
大概就是建出最佳情况以后强行扭一下用最小的代价把x扭进来
具体实现见代码

#include <bits/stdc++.h>
using namespace std;
int n,tot,lend,post,lengthd,m,N,p,q,o,x,y,lord;
int fir[100001],nex[200001],dis[100001],to[200001],len[200001],d[100001],sum_d[100001],rank[100001];
int fa[100001],best[100001],dep[100001],top[100001],ans[100001],sum[100001];
void add(int p,int q,int o)
{
    nex[++N]=fir[p];len[N]=o;to[N]=q;fir[p]=N;
}
void Dfs(int now)
{
    for(int i=fir[now];i;i=nex[i])
    if(!dis[to[i]])
    {
        dis[to[i]]=dis[now]+len[i];
        Dfs(to[i]);
    }
}
void dfs(int now)
{
    for(int i=1;i<=n;i++)
        dis[i]=0;
    dis[now]=1;
    Dfs(now);
}
void build(int now,int fat,int deep)
{
    fa[now]=fat;best[now]=deep;dep[now]=deep;rank[now]=lord;
    for(int i=fir[now];i;i=nex[i])
    if(to[i]!=fat)
    {
        build(to[i],now,deep+len[i]);
        best[now]=max(best[now],best[to[i]]);
    }
}
void pou(int now,int Top)
{
    bool sad=1;top[now]=Top;
    for(int i=fir[now];i;i=nex[i])
    if(to[i]!=fa[now])
        if(sad && best[now]==best[to[i]])
            sad=0,pou(to[i],Top);
        else
            pou(to[i],to[i]);
    if(sad)
        ans[++tot]=dep[now]-dep[fa[top[now]]];
}
void find_d()
{
    dfs(1);int start=0,bes=0,end=0;
    for(int i=1;i<=n;i++)
        if(dis[i]>bes) bes=dis[i],start=i;
    dfs(start);bes=0;
    for(int i=1;i<=n;i++)
        if(dis[i]>bes) bes=dis[i],end=i;
    for(d[lend=1]=post=end;post!=start;d[++lend]=post,rank[post]=lend)
        for(int i=fir[post];i;i=nex[i])
            if(dis[to[i]]<dis[post])
            {
                post=to[i];
                break;
            }
    lengthd=dis[end]-1;
    for(int i=2;i<=lend;i++)
        sum_d[i]=dis[d[i]]-1;
}
int main()
{
    scanf("%d%d",&n,&m);
    for(int i=1;i<n;i++)
        scanf("%d%d%d",&p,&q,&o),
        add(p,q,o),add(q,p,o);
    find_d();
    for(int i=1;i<=lend;i++)
        for(int j=fir[d[i]];j;j=nex[j])
        {
            if(i>1 && to[j]==d[i-1]) continue;
            if(i<lend && to[j]==d[i+1]) continue;
            lord=i;
            build(to[j],d[i],len[j]);
            pou(to[j],to[j]);
        }
    sort(ans+1,ans+tot+1,greater<int>());
    for(int i=1;i<=tot;i++)
        sum[i]=sum[i-1]+ans[i];
    int lastans=0;
    for(int i=1;i<=m;i++)
    {
        scanf("%d%d",&x,&y);
        x=(x+lastans-1)%n+1;y=(y+lastans-1)%n+1;
        y=y*2-2;
        if(y>=tot)
        {
            lastans=sum[tot]+lengthd;
            printf("%d\n",lastans);
        }
        else
        if(!y)
            if(!dep[x])
            {
                lastans=lengthd;
                printf("%d\n",lastans);
            }
            else
            {
                lastans=best[x]+max(sum_d[rank[x]],lengthd-sum_d[rank[x]]);
                printf("%d\n",lastans);
            }
        else
        if(dep[x] && best[x]-dep[fa[top[x]]]<ans[y])
        {
            lastans=lengthd+sum[y-1];
            int enter=x;
            while(dep[enter] && best[enter]-dep[fa[top[enter]]]<ans[y])
                enter=fa[top[enter]];
            if(dep[enter])
                lastans+=max(best[x]-dep[enter],best[x]-best[enter]+ans[y]);
            else
            {
                lastans+=best[x];
                if(ans[y]>min(sum_d[rank[x]],lengthd-sum_d[rank[x]]))
                    lastans+=ans[y]-min(sum_d[rank[x]],lengthd-sum_d[rank[x]]);
            }
            printf("%d\n",lastans);
        }
        else
        {
            lastans=sum[y]+lengthd;
            printf("%d\n",sum[y]+lengthd);
        }
    }
    return 0;
}
