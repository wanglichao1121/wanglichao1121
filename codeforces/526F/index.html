codeforces526F

传送门

题意

在一个排列中找有几个子区间满足权值连续

思路

有两套做法啊
首先是套路做法
这种子区间数数第一反应应该是分治
对于每次枚举的中点都能找到所对应的答案区间
然后是我的做法
暴枚一个端点，推移另一个端点，并维护最大最小值
最后统计满足长度==权值范围的区间数
由于是个排列正确性显然

#include <bits/stdc++.h>
#define mid (l+r>>1)
using namespace std;
long long dep,ret,n,p,debug;
int Min[2400001],flag[2400001],num[2400001];
int posa[600001],posb[600001],a[600001],b[600001],s[600001];
void add(int now,int l,int r,int x,int y,long long z)
{
    if(l==x && r==y)
    {
        Min[now]+=z;
        flag[now]+=z;
        return;
    }
    if(x<=mid) add(now<<1,l,mid,x,min(y,mid),z);
    if(y>mid) add(now<<1|1,mid+1,r,max(x,mid+1),y,z);
    if(Min[now<<1]==Min[now<<1|1])
        Min[now]=Min[now<<1]+flag[now],num[now]=num[now<<1]+num[now<<1|1];
    else
    {
        int t;
        if(Min[now<<1]<Min[now<<1|1]) t=now<<1;
            else t=now<<1|1;
        Min[now]=Min[t]+flag[now];
        num[now]=num[t];
    }
}
void query(int now,int l,int r,int x,int y)
{
    if(l==x && r==y)
    {
        if(Min[now]+dep==0)
            ret+=num[now];
        return;
    }
    dep+=flag[now];
    if(x<=mid) query(now<<1,l,mid,x,min(y,mid));
    if(y>mid) query(now<<1|1,mid+1,r,max(x,mid+1),y);
}
void build(int now,int l,int r)
{
    if(l==r)
    {
        Min[now]=0;
        num[now]=1;
        return;
    }
    build(now<<1,l,mid);
    build(now<<1|1,mid+1,r);
    Min[now]=0;
    num[now]=r-l+1;
}
int main()
{
    scanf("%d",&n);
    build(1,1,n);
    for(int i=1;i<=n;i++)
        scanf("%d",&p),scanf("%d",&s[p]);
    int la=0,lb=0;
    posa[0]=n+1;posb[0]=n+1;
    for(int i=n;i>=1;i--)
    {
        while(la && a[la]<=s[i])
        {
            add(1,1,n,posa[la],posa[la-1]-1,s[i]-a[la]);
            --la;
        } 
        while(lb && b[lb]>=s[i])
        {
            add(1,1,n,posb[lb],posb[lb-1]-1,b[lb]-s[i]);
            --lb;
        }
        ++la;++lb;
        a[la]=s[i];posa[la]=i;
        b[lb]=s[i];posb[lb]=i;
        dep=0;
        query(1,1,n,i,n);
        add(1,1,n,i,n,-1);
    }
    printf("%lld\n",ret);
    return 0;
}
