SAM

学习了一下后缀自动机，感觉就这个extend是关键
这里要注意的是fa（后缀连接）指向的是【最长的，Right集合和自己不同的，自己的后缀】
在字符串的末尾加一个字符显然出现了新的Right集合，所以新加一个点，并在上一个最终节点的parent链上找爸爸
（十分类似AC自动机的跳fail？）一路找上去直到有相同的转移边，就找到了爸爸（否则爸爸是空节点1）
然而他的爸爸可能在所在的parent链中被隐藏了，所以可能需要新建出来，然后复制他原来儿子（如果被隐藏，显然只有一个儿子）的转移边，重新确立父子关系，然后回到p那半边去把转移边修一修
这里很容易误解的是从parent边的角度看，这里的p和q是属于完全不同的两条路的，并没有什么关系，q的最后一个字母（最上面的祖先）应该是w，因为是从p由w转移边转移而来，而p就不知道了

char s[N];
struct pam
{
    int cnt,last;
    int a[N][26],fa[N],l[N],size[N];
    pam(){cnt=1;fa[0]=fa[1]=1;l[1]=-1;}
    void extend(int c,int n)
    {
        int p=last;
        while(s[n-l[p]-1]!=s[n])p=fa[p];
        if(!a[p][c])
        {
            int now=++cnt,k=fa[p];
            l[now]=l[p]+2;
            while(s[n-l[k]-1]!=s[n])k=fa[k];
            fa[now]=a[k][c];a[p][c]=now;
        }
        last=a[p][c];
        size[last]++;
    }
    void solve()
    {
        for(int i=cnt;i;i--)
        {
            size[fa[i]]+=size[i];
            ans=max(ans,1ll*size[i]*l[i]);
        }
    }
} pam;
