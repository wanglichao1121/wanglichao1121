传送门
远古的时候学习lct敲的模板题就是这道
然而完全忘光，这次过来重学了一下
维护子树信息的lct感觉还是很兹磁的
写错的几个地方：
1.在access的时候每个被选中的节点都会把它在splay上的右儿子（相当于原树上该节点所在链的后半截）和他的某个虚儿子交换
所以虚儿子信息相当于变了两次
2.在link的时候两个点都要makeroot一下，不然辅助子树信息很难维护，相当于要一路跳父亲上去更新
（由于makeroot最后splay了当前点，所以这个当前点会同时变成原树和辅助树的根）
3.辅助子树信息是两个splay儿子和虚儿子信息的和，一开始很容易想错变成原树的子树信息
（事实上原树基本上没啥用了）
其实维护子树信息的核心想法就是让“辅助子树信息”和“虚儿子信息”互相更新
处理询问的时候肯定先makeroot真的根，然后access询问的子树的根，最后把子树根的信息和他的虚儿子信息合并
（access后他的所有儿子都变成虚儿子了）
（不是很懂为什么直接合并splay右儿子、自身、虚儿子为啥错了，本地跑不过对拍）

#include <bits/stdc++.h>
#define N 200001
using namespace std;
int n,m,CNT,ALL,testid,len;
int X[N],Y[N],Z[N];
bool rev[N];
int c[N][2],sum[N],tr[N],xu[N],fa[N],lis[N],all[N];
void down(int x)
{
    if(rev[x])
    {
        rev[c[x][0]]^=1;rev[c[x][1]]^=1;
        swap(c[x][0],c[x][1]);
        rev[x]=0;
    }
}
void up(int x)
{
    sum[x]=tr[x]^sum[c[x][0]]^sum[c[x][1]];
    all[x]=tr[x]^all[c[x][0]]^all[c[x][1]]^xu[x];
}
bool isroot(int x)
{
    return c[fa[x]][0]!=x && c[fa[x]][1]!=x;
}
void rot(int x)
{
    int y=fa[x],k=c[y][1]==x;
    if(!isroot(y)) c[fa[y]][c[fa[y]][1]==y]=x;
    fa[x]=fa[y];fa[y]=x;
    if(c[x][!k])
        fa[c[x][!k]]=y;
    c[y][k]=c[x][!k];
    c[x][!k]=y;up(y);up(x);
}
void splay(int x)
{
    lis[len=1]=x;
    for(int y=x;!isroot(y);y=fa[y])
        lis[++len]=fa[y];
    while(len)
        down(lis[len--]);
    for(int y;!isroot(x);rot(x))
        if(!isroot(y=fa[x]))
            rot(c[fa[y]][1]==y^c[y][1]==x?x:y);
}
void access(int x)
{
    for(int y=0;x;y=x,x=fa[x])
    {
        splay(x);
        xu[x]^=all[c[x][1]]^all[y];
        c[x][1]=y;up(x);
    }
}
void makeroot(int x)
{
    access(x);
    splay(x);
    rev[x]^=1;
}
void link(int x,int y)
{
    makeroot(x);
    makeroot(y);
    c[y][1]=x;fa[x]=y;
    up(y);
}
void cut(int x,int y)
{
    makeroot(x);
    access(y);
    splay(x);
    fa[y]=0;c[x][c[x][1]==y]=0;
    up(x);
}
void change(int x,int y)
{
    access(x);
    splay(x);
    tr[x]^=y;
    up(x);
}
int que(int x,int rt)
{
    makeroot(rt);
    access(x);
//    down(x);
//    int ans=xu[x]^tr[x];
//    if(c[x][1]) ans^=all[c[x][1]];
//    return ans;
    return tr[x]^xu[x];
}
int main()
{
    srand(time(0));
    scanf("%d%d%d",&testid,&n,&m);
    for(int i=1;i<n;i++)
    {
        int p,q;
        scanf("%d%d",&p,&q);
        link(p,q);
    }
    CNT=0;ALL=0;
    for(int i=1;i<=m;i++)
    {
        int opt,x,y,u,v;
        scanf("%d%d",&opt,&x);
        if(opt==3)
            change(X[x],Z[x]),change(Y[x],Z[x]),ALL^=Z[x];
        else
        {
            scanf("%d",&y);
            if(opt==1)
            {
                scanf("%d%d",&u,&v);
                cut(x,y);link(u,v); 
            }
            else
            if(opt==2)
            {
                ALL^=Z[++CNT]=rand(),change(X[CNT]=x,Z[CNT]),change(Y[CNT]=y,Z[CNT]);
        //        printf("%d\n",Z[CNT]);
            }
            else
            {
                int ans=que(y,x);
        //        printf("%d\n",ans);
                puts(ans==ALL?"YES":"NO");
            }
        }
    }
    return 0;
}
