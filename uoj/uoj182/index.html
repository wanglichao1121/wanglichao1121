传送门
终于写了人生第一发多项式多点求值
事实证明递归fft的效率简直了。。。
此题核心就是计算
可以自己推一下，算出这个东西以后剩下的就是一次函数了
注意两个特判：
1.不满足这个形式，也就是分母上没东西
2.第二个点，全是1，也用不到这坨东西
这个东西其实就可以通过这个多项式的导数除以自身算，
而这个式子可以分治fft算
最后因为询问很多，需要多点求值
写了一天，一半意识模糊写bug，一半意识清醒修bug

#include <bits/stdc++.h>
#define MOD 998244353
#define gg 3
#define si sizeof(int)
#define mid (l+r>>1)
#define N 800001
#define clz(n) __builtin_clz(n)
using namespace std;
int n,m,cnt;
int ret[N],ans[N],g[N];;
int a[N],inv[N],f[N],q[N],w[N],e[N];
int poly[5000001],st[N];
int mi(int x,int y)
{
    int ret=1;
    for(;y;x=1ll*x*x%MOD,y>>=1)
    if(y&1) ret=1ll*ret*x%MOD;
    return ret;
}
int lgn_,n_,rev_[N],omg_[N];
void FFT_init(int n){
    lgn_=32-clz(n-1);n_=1<<lgn_;
    rev_[0]=0;omg_[0]=1;
    long long d=mi(gg,119<<23-lgn_);
    for(int i=1;i<n_;i++){
        omg_[i]=omg_[i-1]*d%MOD;
        rev_[i]=rev_[i/2]>>1|(i&1)<<lgn_-1;
    }
}
long long ww[N];
void FFT(int*a,int flag){
    for(int i=0;i<n_;i++)if(rev_[i]<i)swap(a[i],a[rev_[i]]);
    register int s,t;
    for(int i=0;i<lgn_;i++){
        t=1<<i;
        for(register int j=0;j<1<<i;++j)ww[j]=omg_[j<<lgn_-i-1];
        for(register int j=0;j<n_;j+=t<<1){
            register int*at=a+j,*bt=a+j+t;
            for(register int k=0;k<t;++k){
                s=ww[k]*bt[k]%MOD;
                bt[k]=at[k]-s<0?at[k]-s+MOD:at[k]-s;
                at[k]=at[k]+s>=MOD?at[k]+=s-MOD:at[k]+=s;
            }
        }
    }
    if(flag)
    {
        for(int i=1;i<n_-i;i++)swap(a[i],a[n_-i]);
        long long iv=mi(n_,MOD-2);
    }

}
void dft(int* a,int len,bool inv)
{
    FFT_init(len);
    FFT(a,inv);
}

void fft(int* a,int* b,int len)
{
    dft(a,len,0);dft(b,len,0);
    for(int i=0;i<len;i++)
        a[i]=1ll*a[i]*b[i]%MOD;
    dft(a,len,1);
    for(int i=0;i<len;i++)
        a[i]=1ll*a[i]*inv[len]%MOD;
}
void build(int l,int r,int* f)
//T(n)=O(nlogn)+2*T(n/2)=O(nlog^2n)
{
    if(l==r)
    {
        f[0]=a[l];
        f[1]=1;
        return;
    }
    int le=1;
    while(le<=r-l+2) le<<=1;
    int p[le],q[le];
    build(l,mid,p);build(mid+1,r,q);
    memset(p+mid-l+2,0,si*(le-(mid-l+2)));
    memset(q+r-mid+1,0,si*(le-(r-mid+1)));
    fft(p,q,le);
    memcpy(f,p,si*(r-l+2));
}
void getpoly(int now,int l,int r)
//T(n)=O(nlog^2n)
{
    if(l==r)
    {
        poly[st[now]=++cnt]=MOD-e[l];
        poly[++cnt]=1;
        return;
    }
    int le=1;
    while(le<=r-l+2) le<<=1;
    int p[le],q[le];
    getpoly(now<<1,l,mid);getpoly(now<<1|1,mid+1,r);
    memcpy(p,poly+st[now<<1],si*(mid-l+2));
    memcpy(q,poly+st[now<<1|1],si*(r-mid+1));
    memset(p+mid-l+2,0,si*(le-(mid-l+2)));
    memset(q+r-mid+1,0,si*(le-(r-mid+1)));
    fft(p,q,le);
    memcpy(poly+(st[now]=++cnt),p,si*(r-l+2));
    cnt+=r-l+1;
}
void rev(int* a,int len)
//T(n)=O(nlogn)+T(n/2)=O(nlogn)
{
    if(len==1)
    {
        a[0]=mi(a[0],MOD-2);
        return;
    }
    int h=len/2,d=len*2;
    int p[d],q[d];
    memcpy(p,a,si*h);
    memset(p+h,0,si*(h+len));
    memcpy(q,a,si*len);
    memset(q+len,0,si*len);
    rev(p,h);
    dft(p,d,0);dft(q,d,0);
    for(int i=0;i<d;i++)
        p[i]=((MOD-1ll)*p[i]%MOD*q[i]%MOD+2)*p[i]%MOD;
    dft(p,d,1);
    for(int i=0;i<d;i++)
        p[i]=1ll*p[i]*inv[d]%MOD;
    memcpy(a,p,si*len);
}
void R(int* a,int len)
{
    for(int i=0;i<len>>1;i++)
        swap(a[i],a[len-1-i]);
}
void div(int* p,int l1,int* q,int l2)//l1>=l2
//T(n,m)=O(nlogn)
{
    int nmi=1,divdeg=l1-l2+1;
    while(nmi<max(divdeg*2,l1)) nmi<<=1;
    int x[nmi],y[nmi];
    memcpy(x,p,si*l1);memcpy(y,q,si*l2);
    memset(x+l1,0,si*(nmi-l1));
    memset(y+l2,0,si*(nmi-l2));
    R(x,l1);R(y,l2);
    rev(y,nmi);
    memset(x+divdeg,0,si*(nmi-divdeg));
    memset(y+divdeg,0,si*(nmi-divdeg));
    fft(x,y,nmi);
    memcpy(p,x,si*divdeg);
    R(p,divdeg);memset(p+divdeg,0,si*(l1-divdeg));
}
void mod(int* p,int l1,int* q,int l2)
//T(n,m)=O(nlogn)
{
    if(l1<l2) return;
    int alllen=1;
    while(alllen<l1) alllen<<=1;
    int x[alllen],y[alllen];
    memcpy(x,p,si*l1);memcpy(y,q,si*l2);
    memset(x+l1,0,si*(alllen-l1));
    memset(y+l2,0,si*(alllen-l2));
    div(x,l1,y,l2);
    fft(x,y,alllen);
    for(int i=0;i<l1;i++)
         p[i]=(p[i]-x[i]+MOD)%MOD;
}
void getval(int now,int l,int r,int* f,int len)
//T(n,m)=2*T(n/2,n)+O(mlogm)
//T(1,m)=O(m)
{
    int le=r-l+2;
    int p[le],q[len];
    memcpy(p,poly+st[now],si*le);
    mod(f,len,p,le);
    if(l==r)
    {
        ret[l]=f[0];
        return;
    }
    memcpy(q,f,si*len);
    getval(now<<1,l,mid,f,le);
    memcpy(f,q,si*len);
    getval(now<<1|1,mid+1,r,f,le);
}
int main()
{
    inv[1]=1;
    for(int i=2;i<=600000;i++)
        inv[i]=(MOD-MOD/i)*inv[MOD%i]%MOD;
    scanf("%d%d",&n,&m);
    int sum=0;
    for(int i=1;i<=n;i++)
    {
        scanf("%d",&a[i]);
        sum+=a[i];
        if(sum>=MOD) sum-=MOD; 
    }
    int nmi=1;
    while(nmi<=n) nmi<<=1;
    build(1,n,f);
    for(int i=0;i<n;i++)
        g[i]=1ll*f[i+1]*(i+1)%MOD;
    int z=1,x=0,c=0,v=1,len=0;
    for(int i=1;i<=m;i++)
    {
        int opt;
        scanf("%d",&opt);
        if(opt==1)
        {
            scanf("%d",&opt);
            z+=1ll*c*opt%MOD;x+=1ll*v*opt%MOD;
            if(z>=MOD) z-=MOD;
            if(x>=MOD) x-=MOD;
        }
        else
            swap(z,c),swap(x,v);
        if(c==0)
            ans[i]=(1ll*sum*z%MOD+1ll*x*n%MOD)*mi(v,MOD-2)%MOD;
        else
        {
            int b=1ll*z*mi(c,MOD-2)%MOD,bb=(x-1ll*b*v%MOD+MOD)%MOD;
            int bbb=1ll*v*mi(c,MOD-2)%MOD;
            bb=1ll*bb*mi(c,MOD-2)%MOD;
            ans[i]=-1;
            q[++len]=b;w[len]=bb;e[len]=bbb;
        }
    }
    if(len)
    {
        getpoly(1,1,len);
        getval(1,1,len,f,n+1);
//    cerr << TIME << endl;
        int bf[200001];
        for(int i=1;i<=len;i++)
            bf[i]=ret[i];
        getval(1,1,len,g,n);
        for(int i=1;i<=len;i++)
            ret[i]=1ll*ret[i]*mi(bf[i],MOD-2)%MOD;
    }
    int wlwl=0;
    for(int i=1;i<=m;i++)
    if(ans[i]>0) printf("%d\n",ans[i]);
    else
    {
        ++wlwl;
        printf("%d\n",(1ll*ret[wlwl]*w[wlwl]+1ll*q[wlwl]*n)%MOD);
    }
    return 0;
}
