传送门
论文里给出了一个很牛逼的算法啊
虽然线段树做法也蛮好写的，但是还是体验一下这个离线做法吧
已知n条折线，总拐点数<=m，求上下凸壳
论文里的做法：每次调两条拐点数最少的合并成一条
复杂度证明：哈夫曼树
没了
话说我一开始写了个分数类，然后发现有的地方longlong根本存不了
事实上整除也是兹磁的
话说我一开始没保存k，现场算，跟别的东西配合起来会出现被0除
事实上存一下又没费多少空间
话说我对拍的时候rd时间不单调

#include <bits/stdc++.h>
#define N 600001
using namespace std;
int n,m,p[N],t[N],k[N],x[N],cnt[N],pool1[20*N],pool2[20*N];
long long pool3[20*N],ans[N];bool ch[N];char opt[10];
priority_queue<pair<int,int>,vector<pair<int,int> >,greater<pair<int,int> > > sz;
struct poly
{
    int len,*x,*k;
    long long *y;
    void insert(int T,int K){x[++len]=T;k[len]=K;y[len]=y[len-1]+1ll*k[len-1]*(x[len]-x[len-1]);}
    long long calc(int i,int p){return y[i]+1ll*(p-x[i])*k[i];}
    void insert(int a,long long b,int c){x[++len]=a,y[len]=b,k[len]=c;}
    void merge(poly *a,poly *b)
    {
        int t1,t2,t3;
        long long h1,h2;
        len=0;
        for (int i=1,j=1;i<a->len && j<b->len;)
        {
            t1=max(a->x[i],b->x[j]);
            t2=min(a->x[i+1],b->x[j+1])-1;
            if ((h1=a->calc(i,t1))<(h2=b->calc(j,t1)))
                swap(a,b),swap(i,j),swap(h1,h2);
            insert(t1,h1,a->k[i]);
            if (a->calc(i,t2)<b->calc(j,t2))
            {
                t3=(h1-h2)/(b->k[j]-a->k[i])+t1+1;
                insert(t3,b->calc(j,t3),b->k[j]);
            }
            if (a->x[i+1]-1<=t2)i++;
            if (b->x[j+1]-1<=t2)j++;
        }
        x[++len]=1e9;
    }
} a[N];
void solve()
{
    for(int i=1;i<=n;i++)
        a[i].x[++a[i].len]=1e9;
    for(int i=1;i<=n;i++) 
        sz.push(make_pair(a[i].len,i));
    for(int tot=n,alllen=cnt[n];sz.size()>1;sz.push(make_pair(a[tot].len,tot)))
    {
        poly *t1=&a[sz.top().second],*t2;
        sz.pop();t2=&a[sz.top().second];sz.pop();
        a[++tot].x=pool1+alllen,a[tot].k=pool2+alllen,a[tot].y=pool3+alllen;
        a[tot].merge(t1,t2);
        alllen+=a[tot].len;
    }
    for(int i=1,j=1;i<=m;i++)
        if(!ch[i])
        {
            while(a[sz.top().second].x[j+1]<=t[i])j++;
            ans[i]=max(ans[i],a[sz.top().second].calc(j,t[i]));
        }
}
int main()
{
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;i++)
        scanf("%d",&p[i]);
    for(int i=1;i<=m;i++)
    {
        scanf("%d%s",&t[i],opt);
        if (opt[0]=='c')
        {
            int p;
            scanf("%d%d",&p,&x[i]);
            ++cnt[k[i]=p];
            ch[i]=1;
        }
    }
    for(int i=1;i<=n;i++)
    {
        a[i].x=pool1+cnt[i-1],a[i].k=pool2+cnt[i-1],a[i].y=pool3+cnt[i-1];
        a[i].y[a[i].len=1]=p[i],cnt[i]+=cnt[i-1]+2;
    }
    for(int i=1;i<=m;i++)
        if (ch[i])
            a[k[i]].insert(t[i],x[i]);
    solve();
    for(int i=1;i<=n;i++)
        a[i].y[a[i].len=1]=-p[i];
    for(int i=1;i<=m;i++)
        if (ch[i])
            a[k[i]].insert(t[i],-x[i]);
    sz.pop();
    solve();
    for(int i=1;i<=m;i++)
        if(!ch[i])
            printf("%lld\n",ans[i]);
    return 0;
}
