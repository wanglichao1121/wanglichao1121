传送门
一开始听小猫bb了一个复杂度错误的算法我还信以为真。。。

题意

维护一个序列
要求支持：
1.区间加
2.区间开根号，下取整
3.区间求和

思路

如果没有区间加的话可以发现差不多loglog次以后就全变1了
有区间加以后这个性质没了，但是考虑到区间加对差分后序列的影响是O(1)的，我们可以找差分序列上的较弱性质。首先发现在很多情况下：

有一个感性理解就是用类似平方差公式的东西去bb

但是具体没推过。。。不是很懂这个1对答案影响有多大
我们不妨先假装没有影响（雾）
那么每次都会变成原来的根号级别（玄）
于是在loglog次以后也没了
由于每次区间加对这个东西的影响都是O(1)的，我们不妨也忽略掉（雾）
那么这个复杂度就对啦！！！（雾）
反正相邻两数差在开平方以后的确变小不少
于是我就有一个思路：维护差分序列。然而这样子的话询问就变成了二阶求和。。。浑身难受
小猫bb的错误黑科技：直接维护原数列，开根号的时候在线段树上暴力往下跑，如果到一个节点发现这个节点下面所有的数都已经相等了就不跑下去，当成区间减（加一个负数）打个标记就好了
↑我第一遍听小猫bb的想法↑
码完一交，发现T了
打开小猫程序，发现还有一段东西。。。
注意前面的复杂度分析不但有很多符合直觉但是不是很严谨的地方，还有一个加粗的在很多情况下。
举个例子：如果一个区间只有3和4，那么开完根号所有3变1,4变2，差并没有减小

大力分析一波：如果每一次开根号操作都能让差分后序列减小的话，上面的复杂度分析可以假装是对的
但是如果遇上退化成区间加（区间内只有和）的开根号就会炸
所以我们特判掉就做完了

#include <bits/stdc++.h>
#define mid (l+r>>1)
using namespace std;
int n,m,a[500001],x,l,r,cmd;
long long mi[500001],ma[500001],sum[500001],flag[500001];
void upd(int now)
{
    mi[now]=min(mi[now<<1],mi[now<<1|1]);
    ma[now]=max(ma[now<<1],ma[now<<1|1]);
    sum[now]=sum[now<<1]+sum[now<<1|1];
}
void inc(int now,int l,int r,long long x)
{
    mi[now]+=x;ma[now]+=x;
    sum[now]+=(long long)(r-l+1)*x;
    flag[now]+=x;
}
void down(int now,int l,int r)
{
    if(flag[now])
        inc(now<<1,l,mid,flag[now]),
        inc(now<<1|1,mid+1,r,flag[now]),
        flag[now]=0;
}
void add(int now,int l,int r,int x,int y,int z)
{
    if(l==x && r==y)
    {
        inc(now,l,r,z);
        return;
    }
    down(now,l,r);
    if(x<=mid) add(now<<1,l,mid,x,min(y,mid),z);
    if(y>mid) add(now<<1|1,mid+1,r,max(x,mid+1),y,z); 
    upd(now);
}
void rua(int now,int l,int r,int x,int y)
{
    if(l==x && r==y)
    {
        long long tem=sqrt(ma[now]);
        if(ma[now]==tem*tem && mi[now]==tem*tem-1 || mi[now]==ma[now])
        {
            inc(now,l,r,(int)sqrt(mi[now])-mi[now]);
            return;
        }
    }
    down(now,l,r);
    if(x<=mid) rua(now<<1,l,mid,x,min(y,mid));
    if(y>mid) rua(now<<1|1,mid+1,r,max(x,mid+1),y);
    upd(now);
}
void build(int now,int l,int r)
{
    if(l==r)
    {
        mi[now]=ma[now]=sum[now]=a[l];
        flag[now]=0;
        return;
    } 
    build(now<<1,l,mid);build(now<<1|1,mid+1,r);
    upd(now);
}
long long que(int now,int l,int r,int x,int y)
{
    if(l==x && r==y) return sum[now];
    down(now,l,r);
    long long ret=0;
    if(x<=mid) ret+=que(now<<1,l,mid,x,min(y,mid));
    if(y>mid) ret+=que(now<<1|1,mid+1,r,max(mid+1,x),y);
    return ret;
}
int main()
{
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;i++)
        scanf("%d",&a[i]);
    build(1,1,n);
    for(int i=1;i<=m;i++)
    {
        scanf("%d%d%d",&cmd,&l,&r);
        if(cmd==1)
        {
            scanf("%d",&x);
            add(1,1,n,l,r,x);
        }
        else
        if(cmd==2)
            rua(1,1,n,l,r);
        else
            printf("%lld\n",que(1,1,n,l,r));
    }
    return 0;
}
