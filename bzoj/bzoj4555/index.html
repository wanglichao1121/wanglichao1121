传送门
给定n求
设，结合梁大的bb，可以得到的递推式，然后构造，可以发现，然后一通操作把h看做生成函数，可以发现他和相乘之积+1就是他自身
所以我们把p取个反加个1取个逆乘个阶乘求个前缀和就做完啦

#include <bits/stdc++.h>
#define MOD 998244353
#define g 3
using namespace std;
int n,inv[800001],a[400001];
int mi(int x,int y)
{
    int ret=1;
    for(;y;x=1ll*x*x%MOD,y>>=1)
    if(y&1) ret=1ll*ret*x%MOD;
    return ret;
}
void dft(int* a,int len,bool inv=0)
{
    if(len==1)
        return;
    int x[len/2],y[len/2];
    for(int i=0;i<len/2;i++)
        x[i]=a[2*i],y[i]=a[2*i+1];
    dft(x,len/2,inv);dft(y,len/2,inv); 
    int w=mi(g,(MOD-1)/len),wn=1;
    if(inv)
        w=mi(w,MOD-2);
    for(int i=0;i<len/2;i++,wn=1ll*wn*w%MOD)
    {
        a[i]=x[i]+1ll*wn*y[i]%MOD;
        if(a[i]>=MOD) a[i]-=MOD;
        a[i+len/2]=x[i]-1ll*wn*y[i]%MOD;
        if(a[i+len/2]<0) a[i+len/2]+=MOD;
    }
}
void rev(int* a,int len)
{
    if(len==1)
    {
        a[0]=mi(a[0],MOD-2);
        return;
    }
    int h=len>>1,d=len<<1;
    int tem[d],ha[d];
    memcpy(tem,a,sizeof(int)*len);
    memset(tem+len,0,sizeof(int)*len);
    memcpy(ha,a,sizeof(int)*h);
    rev(ha,h);
    memset(ha+h,0,sizeof(int)*(d-h));
    dft(tem,d);dft(ha,d);
    for(int i=0;i<d;i++)
    {
        tem[i]=(MOD-1ll)*tem[i]%MOD*ha[i]%MOD+2;
        if(tem[i]>=MOD) tem[i]-=MOD;
        tem[i]=1ll*tem[i]*ha[i]%MOD;
    }
    dft(tem,d,1);
    for(int i=0;i<len;i++)
        a[i]=1ll*tem[i]*inv[d]%MOD;
} 
int main()
{
    scanf("%d",&n);
    inv[1]=1;
    for(int i=2;i<=4*n;i++)
        inv[i]=1ll*(MOD-MOD/i)*inv[MOD%i]%MOD;
    a[1]=2;
    for(int i=2;i<=n;i++)
        a[i]=1ll*a[i-1]*inv[i]%MOD;
    for(int i=1;i<=n;i++)
        a[i]=MOD-a[i];
    a[0]=1;
    int nmi=1;
    while(nmi<=n) nmi<<=1;
    rev(a,nmi);
    int ans=0,jc=1;
    for(int i=0;i<=n;jc=1ll*jc*++i%MOD)
        ans=(ans+1ll*a[i]*jc)%MOD;
    printf("%d\n",ans);
    return 0;
}
